import { ReactLenis } from 'lenis/react';
import { ReactNode } from 'react';

export function SmoothScroll({ children }: { children: ReactNode }) {
	return (
		<ReactLenis root options={{ lerp: 0.1, duration: 1.5, smoothWheel: true }}>
			{children}
		</ReactLenis>
	);
}

### Purpose & Flow:
This component wraps its children with a smooth scrolling effect using the Lenis library. It is used in the root layout to provide a modern, smooth scroll experience across all pages. All page content is rendered as children inside this component, ensuring consistent scroll behavior throughout the app.

## src/components/CustomCursor.tsx

### Code:
'use client';

import { motion, useMotionValue, useSpring } from 'framer-motion';
import { useEffect, useState } from 'react';

export default function CustomCursor() {
	const [isHovered, setIsHovered] = useState(false);
	const cursorX = useMotionValue(-100);
	const cursorY = useMotionValue(-100);

	const springConfig = { damping: 25, stiffness: 400 };
	const cursorXSpring = useSpring(cursorX, springConfig);
	const cursorYSpring = useSpring(cursorY, springConfig);

	useEffect(() => {
		const moveCursor = (e: MouseEvent) => {
			cursorX.set(e.clientX);
			cursorY.set(e.clientY);
		};

		const handleHover = (e: MouseEvent) => {
			const target = e.target as HTMLElement;
			setIsHovered(!!target.closest('button, a, .cursor-pointer'));
		};

		window.addEventListener('mousemove', moveCursor);
		window.addEventListener('mouseover', handleHover);

		return () => {
			window.removeEventListener('mousemove', moveCursor);
			window.removeEventListener('mouseover', handleHover);
		};
	}, [cursorX, cursorY]);

	return (
		<motion.div
			className="fixed top-0 left-0 w-8 h-8 border border-gold rounded-full pointer-events-none z-[9999] hidden md:block"
			style={{
				translateX: cursorXSpring,
				translateY: cursorYSpring,
				scale: isHovered ? 2.5 : 1,
				backgroundColor: isHovered ? 'rgba(197, 160, 89, 0.1)' : 'transparent',
			}}
			transition={{ type: 'spring', damping: 20, stiffness: 250 }}
		/>
	);
}

### Purpose & Flow:
This component creates a custom animated cursor that follows the user's mouse and grows when hovering over interactive elements. It is used globally in the layout to enhance the luxury, interactive feel of the site.

'use client';

import React, { useState, useEffect, useMemo } from 'react';
import { X, Plus, ChevronLeft, ChevronRight, Info, ShoppingBag, Search, Grid, BookOpen } from 'lucide-react';
import { motion, AnimatePresence } from 'framer-motion';
import { CATALOG_DATA, Spread, Product } from '@/data/catalog';

export default function Catalog() {
	const [currentSpreadIndex, setCurrentSpreadIndex] = useState(0);
	const [selectedProduct, setSelectedProduct] = useState<Product | null>(null);
	const [viewMode, setViewMode] = useState<'brochure' | 'gallery'>('brochure');
	const [searchQuery, setSearchQuery] = useState('');
	const [isSearchOpen, setIsSearchOpen] = useState(false);
  
	const currentSpread = CATALOG_DATA[currentSpreadIndex];

	// Flat list of all products for search and gallery
	const allProducts = useMemo(() => {
		return CATALOG_DATA.flatMap(spread => spread.products);
	}, []);

	const filteredProducts = useMemo(() => {
		if (!searchQuery) return allProducts;
		return allProducts.filter(p => 
			p.name.toLowerCase().includes(searchQuery.toLowerCase()) || 
			p.category.toLowerCase().includes(searchQuery.toLowerCase())
		);
	}, [searchQuery, allProducts]);

	useEffect(() => {
		if (selectedProduct || isSearchOpen) {
			document.body.style.overflow = 'hidden';
		} else {
			document.body.style.overflow = 'unset';
		}
	}, [selectedProduct, isSearchOpen]);

	const handleNextPage = () => {
		if (currentSpreadIndex < CATALOG_DATA.length - 1) {
			setCurrentSpreadIndex(prev => prev + 1);
		}
	};

	const handlePrevPage = () => {
		if (currentSpreadIndex > 0) {
			setCurrentSpreadIndex(prev => prev - 1);
		}
	};

	return (
		<div className="bg-ivory text-obsidian overflow-hidden selection:bg-gold/20 font-sans">
			{/* CATALOG SUB-NAV */}
			<div className="h-14 border-b border-gray-100 flex justify-between items-center px-8 md:px-20 bg-white/50 backdrop-blur-md sticky top-20 z-30">
				 <div className="flex items-center gap-8">
						<button 
							onClick={() => setViewMode('brochure')}
							className={`flex items-center gap-2 text-[10px] uppercase tracking-widest font-black transition-all ${viewMode === 'brochure' ? 'text-gold' : 'text-gray-400 hover:text-obsidian'}`}
						>
							<BookOpen size={14} /> Brochure
						</button>
						<button 
							onClick={() => setViewMode('gallery')}
							className={`flex items-center gap-2 text-[10px] uppercase tracking-widest font-black transition-all ${viewMode === 'gallery' ? 'text-gold' : 'text-gray-400 hover:text-obsidian'}`}
						>
							<Grid size={14} /> Gallery
						</button>
				 </div>

				 <div className="flex items-center gap-6">
						<button 
							onClick={() => setIsSearchOpen(true)}
							className="flex items-center gap-2 text-[10px] uppercase tracking-widest font-black text-gray-400 hover:text-gold transition-all"
						>
							<Search size={14} /> Search Discovery
						</button>
				 </div>
			</div>

			<AnimatePresence mode="wait">
				{viewMode === 'brochure' ? (
					<motion.div 
						key="brochure"
						initial={{ opacity: 0 }}
						animate={{ opacity: 1 }}
						exit={{ opacity: 0 }}
						className="flex flex-col lg:flex-row relative h-[calc(100vh-136px)] overflow-hidden"
					>
						{/* LEFT PAGE: Visuals */}
						<div className="w-full lg:w-[60%] h-[40vh] lg:h-full relative bg-gray-100 flex-shrink-0 group overflow-hidden">
							<AnimatePresence mode="wait">
								<motion.div
									key={currentSpread.image}
									initial={{ opacity: 0, scale: 1.05 }}
									animate={{ opacity: 1, scale: 1 }}
									exit={{ opacity: 0, scale: 0.98 }}
									transition={{ duration: 1.2, ease: [0.16, 1, 0.3, 1] }}
									className="w-full h-full"
								>
									<img src={currentSpread.image} alt={currentSpread.title} className="w-full h-full object-cover" />
								</motion.div>
							</AnimatePresence>
							{/* Hotspots for products */}
							{currentSpread.products.map((product, idx) => (
								<button
									key={product.id}
									className="absolute z-10"
									style={{ left: product.hotspot.x, top: product.hotspot.y }}
									onClick={() => setSelectedProduct(product)}

									<span className="w-8 h-8 flex items-center justify-center rounded-full bg-gold/80 text-white shadow-lg hover:scale-110 transition-transform">
										<Plus size={18} />
									</span>
								</button>
							))}
						</div>

						{/* RIGHT PAGE: Details */}
						<div className="flex-1 flex flex-col justify-center px-8 md:px-20 py-12 lg:py-0">
							<span className="text-luxury mb-8 block">{currentSpread.pageNumber}</span>
							<h2 className="text-5xl md:text-7xl font-serif mb-12 leading-[1.1]">{currentSpread.title}</h2>
							<p className="text-lg text-gray-500 mb-12 leading-relaxed">{currentSpread.description}</p>

								<button
									onClick={handlePrevPage}
									disabled={currentSpreadIndex === 0}
									className="p-4 rounded-full border border-gold text-gold disabled:opacity-30"
								>
									<ChevronLeft size={24} />

								<button
									onClick={handleNextPage}
									disabled={currentSpreadIndex === CATALOG_DATA.length - 1}
									className="p-4 rounded-full border border-gold text-gold disabled:opacity-30"
								>
									<ChevronRight size={24} />
								</button>
							</div>
						</div>
					</motion.div>
				) : (
					<motion.div
						key="gallery"
						initial={{ opacity: 0 }}
						animate={{ opacity: 1 }}
						exit={{ opacity: 0 }}
						className="p-8 md:p-20 grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-16"
					>
						{filteredProducts.map(product => (
							<div key={product.id} className="bg-white rounded-2xl shadow-xl p-8 flex flex-col gap-6 cursor-pointer hover:shadow-2xl transition-shadow" onClick={() => setSelectedProduct(product)}>
								<div className="aspect-square bg-gray-100 rounded-xl mb-4 flex items-center justify-center">
									<Info size={32} className="text-gold" />
								</div>
								<h3 className="text-2xl font-serif mb-2">{product.name}</h3>
								<p className="text-luxury text-gold mb-2">{product.category}</p>
								<p className="text-gray-500 text-sm mb-2">{product.description}</p>
								<p className="text-xs text-gray-400">{product.materials}</p>
								<p className="text-xs text-gray-400">{product.dimensions}</p>
								<p className="text-lg text-gold font-bold mt-2">${product.price}</p>
							</div>
						))}
					</motion.div>
				)}
			</AnimatePresence>

			{/* Product Modal */}
			<AnimatePresence>
				{selectedProduct && (
					<motion.div
						initial={{ opacity: 0 }}
						animate={{ opacity: 1 }}
						exit={{ opacity: 0 }}
						className="fixed inset-0 bg-black/60 z-50 flex items-center justify-center"
						onClick={() => setSelectedProduct(null)}
					>
						<motion.div
							initial={{ scale: 0.95, opacity: 0 }}
							animate={{ scale: 1, opacity: 1 }}
							exit={{ scale: 0.95, opacity: 0 }}
							transition={{ duration: 0.2 }}
							className="bg-white rounded-2xl p-12 max-w-lg w-full relative"
							onClick={e => e.stopPropagation()}
						>
							<button className="absolute top-6 right-6 text-gray-400 hover:text-gold" onClick={() => setSelectedProduct(null)}>
								<X size={24} />
							</button>
							<h3 className="text-3xl font-serif mb-4">{selectedProduct.name}</h3>
							<p className="text-luxury text-gold mb-2">{selectedProduct.category}</p>
							<p className="text-gray-500 mb-4">{selectedProduct.description}</p>
							<p className="text-xs text-gray-400 mb-2">{selectedProduct.materials}</p>
							<p className="text-xs text-gray-400 mb-2">{selectedProduct.dimensions}</p>
							<p className="text-lg text-gold font-bold mb-4">${selectedProduct.price}</p>
							<button className="w-full py-4 bg-gold text-white rounded-full font-bold flex items-center justify-center gap-2">
								<ShoppingBag size={18} /> Request Purchase
							</button>
						</motion.div>
					</motion.div>
				)}
			</AnimatePresence>

			{/* Search Modal */}
			<AnimatePresence>
				{isSearchOpen && (
					<motion.div
						initial={{ opacity: 0 }}
						animate={{ opacity: 1 }}
						exit={{ opacity: 0 }}
						className="fixed inset-0 bg-black/60 z-50 flex items-center justify-center"
						onClick={() => setIsSearchOpen(false)}
					>
						<motion.div
							initial={{ scale: 0.95, opacity: 0 }}
							animate={{ scale: 1, opacity: 1 }}
							exit={{ scale: 0.95, opacity: 0 }}
							transition={{ duration: 0.2 }}
							className="bg-white rounded-2xl p-12 max-w-lg w-full relative"
							onClick={e => e.stopPropagation()}
						>
							<button className="absolute top-6 right-6 text-gray-400 hover:text-gold" onClick={() => setIsSearchOpen(false)}>
								<X size={24} />
							</button>
							<input
								type="text"
								placeholder="Search by name or category..."
								className="w-full p-4 border-b border-gold text-lg mb-8 focus:outline-none"
								value={searchQuery}
								onChange={e => setSearchQuery(e.target.value)}
								autoFocus
							/>
							<div className="max-h-64 overflow-y-auto">
								{filteredProducts.map(product => (
									<div key={product.id} className="p-4 border-b border-gray-100 cursor-pointer hover:bg-gold/10" onClick={() => { setSelectedProduct(product); setIsSearchOpen(false); }}>
										<h4 className="text-lg font-serif">{product.name}</h4>
										<p className="text-xs text-gray-400">{product.category}</p>
									</div>
								))}
								{filteredProducts.length === 0 && <p className="text-gray-400">No products found.</p>}
							</div>
						</motion.div>
					</motion.div>
				)}
			</AnimatePresence>
		</div>
	);
}

# Application Flow

- The entry point is `src/app/layout.tsx`, which sets up global styles, fonts, and wraps all pages with `SmoothScroll` and `CustomCursor`.
- Each route in `src/app/` (e.g., `page.tsx`, `philosophy/page.tsx`, etc.) represents a page and imports shared components like `Header` and `Footer`.
- Components in `src/components/` are used across pages for UI and interactivity.
- `src/data/catalog.ts` provides product and catalog data for the `Catalog` component.
- The flow is: Layout → Page → Components (Header, Footer, etc.) → Data (if needed).
